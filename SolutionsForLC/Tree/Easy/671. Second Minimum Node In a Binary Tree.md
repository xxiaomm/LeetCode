class Solution {
    int min = -1, res = -1;
    public int findSecondMinimumValue(TreeNode root) {
        if (root == null || (root.left == null && root.right == null))
            return -1;
        min = root.val;
        res = min;
        dfs(root);
        return res == min ? -1 : res;
    }
    public void dfs(TreeNode root) {
        if (root == null) return;
        if (root.val > min) 
            res = res == min ? root.val : Math.min(res, root.val);
        dfs(root.left);
        dfs(root.right);
    }
}
